CREATE TABLE ESTACIONES(
   ID			  INT PRIMARY KEY,
   NOMBRE         TEXT      NOT NULL,
   DESCRIPCION 	  TEXT 
);
CREATE TABLE INSUMOS(
	ID			SERIAL	PRIMARY KEY,
	NOMBRE		TEXT	NOT NULL,
	UNIDAD_DE_MEDIDA TEXT NOT NULL,
	CANTIDAD_ACTUAL REAL NOT NULL
);
CREATE TABLE CATEGORIA_PRODUCTOS(
	ID 			   INT 		PRIMARY KEY,
    NOMBRE         TEXT      NOT NULL
);
CREATE TABLE PRODUCTOS(
   	ID          SERIAL   PRIMARY KEY,
	ID_CATEGORIA_PRODUCTOS INT NOT NULL,
   	NOMBRE      TEXT     NOT NULL,
   	DESCRIPCION TEXT     NOT NULL,
   	PRECIO      INT 		NOT NULL,
	FOREIGN KEY (ID_CATEGORIA_PRODUCTOS) REFERENCES  CATEGORIA_PRODUCTOS(ID)
);
CREATE TABLE CATEGORIA_PEDIDOS(
	ID			INT PRIMARY KEY,
	NOMBRE         TEXT      NOT NULL
);
CREATE TABLE PEDIDOS(
	ID SERIAL PRIMARY KEY,
	ID_CATEGORIA_PEDIDOS INT NOT NULL,
	PRECIO_TOTAL INT NOT NULL,
	NOMBRE_CLIENTE TEXT,
	TELEFONO_CLIENTE TEXT,
	MESA INT,
	DIRECCION TEXT,
	ESTADO INT,
	FECHA_PEDIDO TIMESTAMP,
	HORA_INICIO TIME,
	HORA_ENTREGA TIME,
	FOREIGN KEY (ID_CATEGORIA_PEDIDOS) REFERENCES CATEGORIA_PEDIDOS(ID)
);
--Creacion Tablas Muchos a Muchos
CREATE TABLE PRODUCTOS_ESTACIONES(
	ID_PRODUCTO INT NOT NULL,
	ID_ESTACION INT NOT NULL,
	FOREIGN KEY (ID_PRODUCTO) REFERENCES PRODUCTOS(ID),
	FOREIGN KEY (ID_ESTACION) REFERENCES ESTACIONES(ID)
);
CREATE TABLE PRODUCTOS_INSUMOS(
	ID_PRODUCTO INT NOT NULL,
	ID_INSUMOS INT NOT NULL,
	CANTIDAD REAL NOT NULL,
	FOREIGN KEY (ID_PRODUCTO) REFERENCES PRODUCTOS(ID),
	FOREIGN KEY (ID_INSUMOS) REFERENCES INSUMOS(ID)
);
CREATE TABLE PRODUCTOS_PEDIDO(
	ID_PRODUCTO INT NOT NULL,
	ID_PEDIDO INT NOT NULL,
	CANTIDAD INT NOT NULL,
	PRECIO_X_CANTIDAD INT NOT NULL,
	OBSERVACIONES TEXT,
	ESTACIONES_OK INT NOT NULL,
	ESTACIONES_TOTALES INT NOT NULL,
	FOREIGN KEY (ID_PRODUCTO) REFERENCES PRODUCTOS(ID),
	FOREIGN KEY (ID_PEDIDO) REFERENCES PEDIDOS(ID)
);
